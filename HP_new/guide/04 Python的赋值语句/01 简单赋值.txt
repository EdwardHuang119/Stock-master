'''
   简单赋值

    命令格式：
           变量名=表达式

    变量名的数据类型根据表达式的运算结果而定。

请看示例。
'''

a=10               #变量a的值是整数10。
b=10.5+a*2.3       #变量b值是表达式的计算结果33.5。
c=[1,2,4,5]        #变量c是一个列表。
d=(1,2,3,4,5)      #变量d是元组。
e=True             #变量e是逻辑变量，值为True真。
f=1.5+2.5j         #变量f是复数。
g='hello Python !' #变量g是字符串 。
print('a= ',a)
print('b= ',b)
print('c= ',c)
print('d= ',d)
print('e= ',e)
print('f= ',f)
print('g= ',g)
